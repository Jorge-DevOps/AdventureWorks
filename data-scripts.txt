EMPLEADOS POR DEPARTAMENTO

SELECT
	D.NAME, COUNT(ED.EMPLOYEEID)
FROM
	HUMANRESOURCES.DEPARTAMENT D
	JOIN HUMANRESOURCES.EMPLOYEEDEPARTMANETHISTORY ED ON D.DEPARTMENTID = ED.DEPARTMENTID
GROUP BY
	D.NAME;

CANTIDAD DE PRODUCTOS EN BODEGA

SELECT
	PP.NAME, PI.QUANTITY
FROM
	PRODUCTION.PRODUCT PP
	JOIN PRODUCTION.PRODUCTINVENTORY PI ON PP.PRODUCTID = PI.PRODUCTID;
